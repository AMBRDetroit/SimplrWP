 71     public function load_template($template) {
 72         global $wp_query, $single_object;
 73         // if the correct query parameter exists, load a template
 74         if ( isset( $wp_query->query_vars[$this->settings['base_slug']] ) ) {
 75             $query_var = explode('/',$wp_query->query_vars[$this->settings['base_slug']]);
 76             // let's assume the first query var is the ID
 77             $id = $query_var[0];
 78             
 79             $url_parameters = array();
 80             if(empty($id)) {
 81                 $url_parameters[$this->settings['list_page_settings']['page_slug'][0]] = 1;
 82             } else {
 83                 // generate key/value pairs from query vars
 84                 for($i=0; $i<sizeof($query_var); $i+=2) {
 85                     $value = (($i+1)==sizeof($query_var)) ? null : $query_var[$i+1];
 86                     $url_parameters[$query_var[$i]] = $value;
 87                 }
 88             }
 89             
 90             // if the value of the query parameter exists, load the single template
 91             if( !empty($id) && !in_array(implode('-',array_keys($url_parameters)), array_keys($this->sub_pages)) ) {
 92                 // get query fields for object
 93                 preg_match_all('/\[([\w-_]*)\]*([\w-_]*)/i', $this->settings['object_slug'], $query_fields);
 94                 $query_keys = $query_fields[1];
 95                 
 96                 // generate regex for getting the values from query var
 97                 $values_regex = '/';
 98                 for($i=0; $i<sizeof($query_keys); $i++ ) {
 99                     $values_regex .= '(.*)' . $query_fields[2][$i];
100                 }
101                 $values_regex .= '/i';
102                 
103                 // get query values for object
104                 preg_match_all($values_regex, $id, $query_values);
105                 array_shift($query_values);
106                 
107                 // if no matches, it means it's an incorrect URL, so redirect to not found template
108                 if(!empty($query_values[0])) {
109                     $query_values = array_map(function($val) { return $val[0]; }, $query_values);
110                     
111                     // create array of query parameters
112                     $query_parameters = array_combine($query_keys, $query_values);
113                     
114                     // get object based on query parameters
115                     $object_query = new \SimplrWP\Core\ObjectQuery($this->settings['object']);
116                     $query_args = array( 'relation' => 'AND' );
117                     foreach($query_parameters as $key => $value) {
118                         $query_args[] = array(
119                             'key' => $key,
120                             'value' => $value
121                         );
122                     }
123                     
124                     // does this object exist in the db?
125                     $results = $object_query->query(array('where_args' => $query_args ));
126                     if(!empty($results)) {
127                         $single_object = $this->settings['object'];
128                         $single_object->set_id_and_retrieve_data($results[0]['id']);
129                         return $this->settings['single_template_file'];
130                     }
131                 }
132                 // object not found, return not found template
133                 return $this->settings['object_not_found_template_file'];
134             // since this is a sub page, render accordingly 
135             } else {
136                 return $this->render_sub_page($url_parameters); 
137             }
138         }
139         return $template;

142     public function generate_object_permalink($id = null) {
143         if($id) {
144             $current_object = $this->settings['object'];
145             $current_object->set_id_and_retrieve_data($id);
146             
147             // get object fields from object slug
148             preg_match_all('/\[([\w-_]*)\]*([\w-_]*)/i', $this->settings['object_slug'], $object_matches);
149             $object_fields = $object_matches[1];
150     
151             // generate regex for getting the values from query var
152             $object_url = $this->settings['base_slug'] . '/';
153             for($i=0; $i<sizeof($object_fields); $i++ ) {
154                 $object_slug_part = ($object_fields[$i] == 'id') ? $current_object->get_id() : $current_object->fields[$object_fields[$i]]->get_value();
155                 $object_url .= strtolower($object_slug_part) . $object_matches[2][$i];
156             }
157                 
158             return home_url($object_url . '/');
159         }
160         return false;

168     private function render_sub_page($url_parameters = array()) {
169         global $object_collection; 
170         
171         if(!empty($url_parameters)) {
172             // get call back for generating the query args
173             $query_settings = $this->sub_pages[implode('-',array_keys($url_parameters))]['prepare_query_callback'];
174             
175             $object_query = new \SimplrWP\Core\ObjectQuery($this->settings['object']);
176             
177             $query_results = $object_query->query(array_replace_recursive($this->sub_page_query_settings, $query_settings($url_parameters)) );
178             
179             // create objects for each result
180             $object_collection = array();
181             foreach($query_results as $key => $current_object) {
182                 $new_object = $this->settings['object']->get_unique_name();
183                 $object_collection[$key] = new $new_object($current_object['id']);
184             }
185             
186             // return the associated list template
187             return $this->sub_pages[implode('-',array_keys($url_parameters))]['template_file'];
188         }
189         return $this->settings['object_not_found_template_file'];
