 36     public function __construct($options = array()) {
 37         $this->options = array_replace_recursive($this->options, $options);
 38         
 39         add_action( 'admin_menu', array(&$this, 'render_menus'), (!$this->options['parent_slug']) ? 10 : 105 );
 40         
 41         add_action( 'add_meta_boxes', array(&$this, 'render_metaboxes') );
 42         
 43         add_action( 'admin_enqueue_scripts', array(&$this, 'load_admin_scripts'));
 44         
 45         add_action( 'admin_notices', array(&$this, 'display_admin_notices') );
 46         
 47         // load ajax call for sortable field, if necessary
 48         if($this->options['admin_list']['sortable_field']) {
 49             add_action('wp_ajax_simplrwp_sortable', array(&$this, 'save_object_sort_order') );
 50         }
 51         
 52         // delete the object
 53         if(isset($_GET['delete']) && !empty($_GET['id'])) {
 54             $this->options['object']->set_id_and_retrieve_data($_GET['id']);
 55             if($this->options['object']->delete()) {
 56                 wp_redirect($_SERVER['PHP_SELF'] . '?page=' . $_GET['page'] . '&deleted');
 57             }
 58         }
 59         
 60         // update/create the object
 61         if(!empty($_POST)) {
 62             if($this->save_data($_POST)) {
 63                 wp_redirect($_SERVER['PHP_SELF'] . '?page=' . $this->options['object']->get_unique_name() . '&id=' . $this->options['object']->get()['id'] . '&updated');
 64             }
 65         }
 66         if($this->options['object'] && isset($_GET['page'])) {
 67             if(isset($_GET['updated']) && $_GET['page'] == $this->options['object']->get_unique_name()) {
 68                 $this->admin_notices['success'][] = $this->options['object']->get_labels()['singular'] . ' successfully saved!';
 69             }
 70             if(isset($_GET['deleted']) && $_GET['page'] == $this->options['object']->get_unique_name()) {
 71                 $this->admin_notices['success'][] = $this->options['object']->get_labels()['singular'] . ' successfully deleted!';
 72             }
 73         }
 74         $this->register_metabox(array(
 75             'id' => 'submitdiv',
 76             'label' => 'Save ' . $this->options['object']->get_labels()['singular'],
 77             'context' => 'side',
 78             'before_fields_html_template' => SIMPLRWP_PATH . 'templates/wp_admin/save_metabox.php'
 79         ));
 80     }

162         // if there are editable fields, load them

165                 $this->options['object']->fields[$field]->wp_admin_render_field();

216     private function save_data($data = array()) {
217         // protect against CSRF
218         $nonce_id = $this->options['object']->get_unique_name() . '-nonce';
219         if (!isset($_POST[$nonce_id]) || !wp_verify_nonce($_POST[$nonce_id], basename(__FILE__)))
220             return false;
221         
222         // first, let's get the current data
223         if(!empty($data['id']))
224             $this->options['object']->set_id_and_retrieve_data($data['id']);
225         
226         // all looks good, let's attempt to save the data
227         $results = $this->options['object']->update($data);
228         if(isset($results['valid']) && !$results['valid']) {
229             $this->admin_notices['error'] = $this->_render_error_messages($results['errors']->errors);
230         }
231         return $results['valid'];
232     }
